将 disabled 属性赋值为 'true/false' 来设置是否禁止点击，并可通过 small 属性来指定大小（默认default可不设置） 和 on／off-value 属性 on开启，off关闭 来设置开启状态，可通过switchCall设置回调[根据使用情况而定]
<template>
  <div slot="header" style="width:100%">
    <div class="sw-list">
        <div slot="title" class="left-s">开关状态</div>
        <mk-switch slot="value" v-model="checked"/>
    </div>
    <div class="sw-list">
        <div slot="title" class="left-s">开关状态</div>
        <mk-switch slot="value" name="" v-model="value" on-value="1" off-value="0" />
    </div>
    <div class="sw-list">
        <div slot="title" class="left-s">开关状态 </div>
        <mk-switch @switchCall="swFun" small slot="value" v-model="checked"/>
    </div>
  </div>
</template>
<script>
    import MkSwitch from './switch'
    export default {
        components: {
            MkSwitch
        },
        data () {
            return {
                checked: true,
                value: '0'
            }
        },
        methods: {
            swFun (value) {
                console.log(value)
            }
        }
    }
</script>
