描述
<template>
  <div slot="header" style="width:100%">
    <div>1</div>
  </div>
</template>
<script>
  import { Actionsheet, ActionsheetItem } from '../components/actionsheet/index.js'
  export default {
    components: {
      Actionsheet,
      ActionsheetItem
    },
    data () {
      return {
        options: [
          {
            value: '1',
            label: '编辑'
          },
          {
            value: '2',
            label: '收藏'
          },
          {
            value: '3',
            label: '分享'
          },
          {
            value: '4',
            label: '删除',
            disabled: true
          }
        ],
        open1: false,
        open2: false,
        open3: false
      }
    },
    methods: {
      actionsheetShow (index) {
        if(index == 1){
            this.open1 = true
        }else if(index == 2){
            this.open2 = true
        }else if(index == 3){
            this.open3 = true
        }
      },
      actionsheetJS () {
        let self = this
          this.$actionsheet({
            title: 'JS调用',
            options: [...this.options]
          }).then((value) => {
            self.handleAction(value)
          }).catch(() => {
            console.log('close')
          })
      },
      handleAction (value) {
        let label = this.options.filter((item) => {
          return item.value === value
        })[0].label
        this.$toast({message: `您点击了“${label}”`})
      }
    }
  }
</script>
