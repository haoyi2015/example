Toast

基本用法

通过基础的属性 [open,type,align] 设置
常规使用...
将 open 属性赋值为 'true/false'，可以启用 toast 布局，并可通过 type 属性来指定 success, error, warn 和 align 属性来指定 top, center, bottom 其中的值来定义子元素样式的排版方式。图标统一使用字体库，不支持图片
<template>
  <div slot="header" style="width:100%">
    <button class="dt-button dt-button--primary" @click="toastShow">单个组件方式使用</button>
    <button class="dt-button dt-button--primary" @click="toastShowClass('top')">top使用</button>
    <button class="dt-button dt-button--primary" @click="toastShowClass('center')">center使用</button>
    <button class="dt-button dt-button--primary" @click="toastShowClass('bottom')">bottom使用</button>
    <button class="dt-button dt-button--success" @click="JstoastShow">JS方式使用</button>
    <br>
    <br>
    <button class="dt-button dt-button--success" @click="toastShowType('success')">success使用</button>
    <button class="dt-button dt-button--danger" @click="toastShowType('error')">error使用</button>
    <button class="dt-button dt-button--warning" @click="toastShowType('warn')">warn使用</button>
    <toast :open.sync="errorOpen" type="error" align="center">常规使用...</toast>
  </div>
</template>
<script>
  import Toast from './toast'
  export default {
    components: {
      Toast
    },
    data () {
      return {
        errorOpen: false
      }
    },
    methods: {
      toastShow () {
        this.errorOpen = true
      },
      toastShowClass (t) {
        this.$toast({
          message: t+'位置显示',
          type: 'default',
          align: t
        })
      },
      toastShowType (t) {
        this.$toast({
          message: '信息提示',
          type: t,
          align: 'center'
        })
      },
      JstoastShow () {
        this.$toast({
          message: 'JS方式使用',
          type: 'default'
        })
      }
    }
  }
</script>
