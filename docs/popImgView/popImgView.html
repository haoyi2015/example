PopImgView

基本用法

通过基础的 1/24 分栏任意扩展组合形成较为复杂的混合布局。
将type 属性赋值为 'flex'，可以启用 flex 布局，并可通过 justify 属性来指定 start, center, end, space-between, space-around 其中的值来定义子元素的排版方式。
<template>
  <div slot="header" style="width:100%">
    <swiper-view
      :imgList="path"
      :lazyImg="gifd">
    </swiper-view>
  </div>
</template>
<script>
  import {Tab, TabItem} from './tab/index'
  export default {
    components: {
      Tab,
      TabItem
    },
    data () {
      return {
        gifd: '',//默认图片完整路径
        path: '' //图片集合
      }
    }
  }
</script>
